    2x+1&=7 & 3y-2&=-5 & -5z+8&=-2 \\

    x&=3 & y&=-1 & z&=2 
\end{align}

% 多行公式共用同一个编号
\begin{equation}
    \begin{aligned}
        2x+1&=7 & 3y-2&=-5 & -5z+8&=-2 \\
        2x&=6 & 3y&=-3 & -5z&=-10 \\
        x&=3 & y&=-1 & z&=2
    \end{aligned}
\end{equation}
